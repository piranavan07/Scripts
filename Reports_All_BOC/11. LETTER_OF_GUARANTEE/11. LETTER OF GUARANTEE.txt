SELECT 
MAS.MASTER_REF,MAS.RELMSTRREF,MAS.PRINAME_L1 Our_Customer,GFP.GFCPNC CIF,GFP.GFCTP1 Ownership_Code,MAS.BHALF_BRN,CAPB.FULLNAME AS ON_BEHALF_OF_BRANCH,Related party (LIFE STAGE FLAG),Purpose Code--REFERENCES AND CUSTOMER DETAISL
MAS.CTRCT_DATE AS LOG_DATE,LEFT(BAS.TIMESTART,10) AS INPUT_DATE,LEFT(BAS.TIMEFINISH,10) AS TRAN_DATE,EXPIRY_DAT,--FCC input Date--DATE
MAS.AMOUNT,MAS.AMT_O_S,Amount in Base Currency,Amount in Base Currency--CURRENCY_DETAILS
BEN_PTY.ADDRESS1 AS COUNTER_PARTY,BEN_PTY.ADDRESS2 AS COUNTER_PARTY_COUNTRY, APP_PTY.SW_BANK AS COUNTER_PARTY_BANK,MST.NOSTRO as Nostro_Bank, --BENI_DETAILS



PRO.CODE AS PRODUCT_CODE,--H.S. Code,CBSL Goods Code--SPECIAL FIELDS
POS.ACCT_NUM,
BASC.CHG_DUE AS Commi_Chargers,POS.POSTINGTYP AS BEN_SHA_OUR--CHARGES

FROM 
MASTER MAS,LCMASTER LCM,

PARTYDTLS APP_PTY,
PARTYDTLS BEN_PTY,
PARTYDTLS ADV_BNK,
PARTYDTLS REM_BNK,

EXEMPL30 EXE,
EXEMPL30 EXE_PRO,
EXEMPL30 EXE_EVE,

PRODTYPE PRO,
BASEEVENT BAS,
GFPF GFP,
CAPF CAPI,
CAPF CAPB,
POSTING POS,

TIDATAITEM TID,MSTRSETTLE MST,GJPF GJP,
EVENTCHG EVE,
BASECHARGE  BASC

WHERE MAS.KEY97=LCM.KEY97
--AND MAS.REFNO_PFIX='ISB'
--AND MAS.STATUS='LIV'

AND LCM.APP_PTY=APP_PTY.KEY97
AND LCM.BEN_PTY=BEN_PTY.KEY97
AND LCM.AVWBNK_PTY=ADV_BNK.KEY97
AND LCM.REIMB_PTY=REM_BNK.KEY97

AND MAS.EXEMPLAR=EXE.KEY97
AND MAS.EXEMPLAR=EXE_PRO.KEY97
AND MAS.EXEMPLAR=EXE_EVE.KEY97
AND PRO.PRODUCT=EXE.KEY97

AND MAS.PRODTYPE=PRO.KEY97 
AND MAS.KEY97=BAS.MASTER_KEY
AND APP_PTY.CUS_MNM=GFP.GFCUS1
AND MAS.INPUT_BRN=CAPI.CABRNM
AND MAS.BHALF_BRN=CAPB.CABRNM

AND BAS.MASTER_KEY=TID.EVENT_KEY
AND TID.KEY97=MST.KEY97
AND MST.NOSTRO=GJP.GJNST

AND BAS.KEY97=POS.KEY97
AND APP_PTY.KEY97=EVE.KEY97
AND MAS.KEY97=BASC.KEY97
;
